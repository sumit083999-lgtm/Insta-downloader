async function processDownload() { const inputUrl = document.getElementById("videoUrl").value.trim(); const fetchBtn = document.getElementById("fetchBtn"); const btnText = document.getElementById("btnText"); const btnIcon = document.getElementById("btnIcon"); const statusMessage = document.getElementById("statusMessage"); const downloadContainer = document.getElementById("downloadContainer"); const finalDownloadBtn = document.getElementById("finalDownloadBtn"); // Reset layout elements statusMessage.classList.add("hidden"); downloadContainer.classList.add("hidden"); if (!inputUrl) { showStatus("Kripya ek valid video URL paste karein!", "text-red-500"); return; } // UI Loading state update fetchBtn.disabled = true; btnText.innerText = "Fetching Video Links..."; btnIcon.className = "fas fa-spinner fa-spin"; try { // Pubic CORS-Bypass API for downloading Instagram content const targetApi = `https://allorigins.win{encodeURIComponent('https://ddinstagram.com' + inputUrl.split('instagram.com/')[1])}`; const response = await fetch(targetApi); if (!response.ok) throw new Error("Network response error."); const rawData = await response.json(); // Agar dynamic redirect match ho jata hai if (rawData && rawData.contents) { // Instagram wrapper redirect process check const directVideoUrl = 'https://ddinstagram.com' + inputUrl.split('instagram.com/')[1]; showStatus("Link mil gaya! Niche click karke download karein.", "text-green-600"); // Setup download interface buttons finalDownloadBtn.href = directVideoUrl; downloadContainer.classList.remove("hidden"); } else { showStatus("Kuch galat hua! Kripya sahi link check karein.", "text-red-500"); } } catch (error) { console.error("Downloader Error:", error); // Fallback option in case proxy limits hit showStatus("Error! Agar link kaam na kare toh SaveFrom.net try karein.", "text-amber-600"); // Backup link redirection direct implementation finalDownloadBtn.href = `https://savefrom.net{encodeURIComponent(inputUrl)}`; downloadContainer.classList.remove("hidden"); } finally { // Restore button interactive state fetchBtn.disabled = false; btnText.innerText = "Fetch Download Link"; btnIcon.className = "fas fa-arrow-right"; } } function showStatus(text, textColorClass) { const statusMessage = document.getElementById("statusMessage"); statusMessage.innerText = text; statusMessage.className = `mt-4 text-sm font-semibold ${textColorClass}`; statusMessage.classList.remove("hidden"); }
